# Es. 5 - Legge DIM elementi di un vettore e li visualizza sulla console (schermo) in ordine inverso
.data
DIM = 5
vett:	.space DIM

.text
.globl main
.ent main

main:
	la	$t2, vett
	and	$t3, $0, $0	# index

lab1:	li	$v0, 12		# Legge un carattere (12)
	syscall			# Aspetta che l'utente digiti un carattere da tastiera

	sb	$v0, ($t2)	# "Store byte" - Salva il carattere
	addi	$t2, $t2, 1
	addi	$t3, $t3, 1	# Update index
	bne	$t3, 5, lab1	# Ripete x5 il codice di lab1

lab2:	addi	$t3, $t3, -1
	addi	$t2, $t2, -1
	
	lb	$a0, ($t2)	# "Load byte" - Prendo il valore contenuto nell'indirizzo puntato da t2 e lo scrive in a0
	li	$v0, 11		# Scrive un carattere (11)
	syscall			# Visualizza a video il codice ascii contenuto nel registro a0
	
	bne	$t3, 0, lab2	# Ripete x5 il codice di lab2
	li	$v0, 10
	syscall

.end main
